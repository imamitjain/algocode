// LRU cache: map[key]*list.Element + list for eviction order
type LRU struct {
    cap   int
    items map[int]*list.Element    // key → element pointer
    order *list.List               // front = most recent
}
type entry struct{ key, val int }

func (c *LRU) Get(key int) (int, bool) {
    if el, ok := c.items[key]; ok {
        c.order.MoveToFront(el)    // mark as recently used
        return el.Value.(*entry).val, true
    }
    return 0, false
}

func (c *LRU) Put(key, val int) {
    if el, ok := c.items[key]; ok {
        el.Value.(*entry).val = val
        c.order.MoveToFront(el)
        return
    }
    if c.order.Len() == c.cap {    // evict oldest
        oldest := c.order.Back()
        c.order.Remove(oldest)
        delete(c.items, oldest.Value.(*entry).key)
    }
    el := c.order.PushFront(&entry{key, val})
    c.items[key] = el
}
Values are stored as any — always type-assert. Pair with a map for O(1) lookup by key (LRU pattern). Never remove an element that belongs to a different list.

// Grid DFS (4-directional)
func dfs(grid [][]int, vis [][]bool, r, c int) {
    if r < 0 || r >= len(grid) || c < 0 || c >= len(grid[0]) || vis[r][c] {
        return
    }
    vis[r][c] = true
    for d := 0; d < 4; d++ {
        dfs(grid, vis, r+dr[d], c+dc[d])
    }
}

// Find minimum in rotated sorted array (= pivot index)
func findMin(nums []int) int {
    lo, hi := 0, len(nums)-1
    for lo < hi {
        mid := lo + (hi-lo)/2
        if nums[mid] > nums[hi] {
            lo = mid + 1   // min is in right half
        } else {
            hi = mid        // min is at mid or left
        }
    }
    return lo  // index of minimum element
}

// Search in rotated array
func search(nums []int, target int) int {
    lo, hi := 0, len(nums)-1
    for lo <= hi {
        mid := lo + (hi-lo)/2
        if nums[mid] == target {
            return mid
        }
        if nums[lo] <= nums[mid] {
            if nums[lo] <= target && target < nums[mid] {
                hi = mid - 1
            } else { lo = mid + 1 }
        } else {
            if nums[mid] < target && target <= nums[hi] {
                lo = mid + 1
            } else { hi = mid - 1 }
        }
    }
    return -1
}

// String palindrome (two pointers)
l, r := 0, len(s)-1

    if s[l] != s[r] {
        return false
    }

}

// Integer palindrome (no string conversion)
if x < 0 || (x%10 == 0 && x != 0) {
    return false
}
rev := 0
for x > rev {
    rev = rev*10 + x%10; x /= 10
}
return x == rev || x == rev/10

// Expand around center (longest palindrome)
func expand(s string, l, r int) int {
    for l >= 0 && r < len(s) && s[l] == s[r] {
        l--; r++
    }
    return r - l - 1
}

const MOD = 1_000_000_007

// Modular exponentiation O(log exp)
func modPow(base, exp, mod int) int {
    result := 1; base %= mod
    for exp > 0 {
        if exp&1 == 1 {
            result = result * base % mod
        }
        base = base * base % mod
        exp >>= 1
    }
    return result
}

Union-Find (DSU)
DSU
parent := make([]int, n)
rank := make([]int, n)
for i := range parent {
    parent[i] = i
}

func find(x int) int {
    if parent[x] != x {
        parent[x] = find(parent[x])  // path compression
    }
    return parent[x]
}

func union(x, y int) bool {
    px, py := find(x), find(y)
    if px == py {
        return false
    }
    if rank[px] < rank[py] {
        px, py = py, px
    }
    parent[py] = px
    if rank[px] == rank[py] {
        rank[px]++
    }
    return true
}
